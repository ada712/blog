    7.页面停留时间+滚动百分比+滑动次数

    Ada 2023/5/16 
    <!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta http-equiv="X-UA-Compatible" content="IE=edge" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
  </head>

  <body>
    <div>主要测试</div>
    <div>停留时间：<span id="count">0</span></div>
    <img
      id="img"
      src=""
      alt=""
      srcset=""
      style="width: 100px; height: 100px; background: pink"
    />
    <div id="LoveID"></div>
  </body>
  <script>
    // 自定义字符串,用于拼接标签
    var loveStr = '';
    // 自定义数组
    // var loveArr = ['喜爱一个人需要理由吗？', '需要吗？', '不需要吗？'];
    /****普通for循环****/
    for (let i = 0; i < 100; i++) {
      loveStr += '<div>' + i + '</div>';
    }
    // 拼接完字符串数组后用innerHTML把它渲染到页面中
    document.getElementById('LoveID').innerHTML = loveStr;

    // -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- --以下是参数获取-- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -- -

    // 1.计算用户滑动次数----------------------------------------------------------------------
    let slider = 0;
    document.body.addEventListener('touchend', () => {
      slider++;
    });

    // 2.计算页面滚动百分比------------------------------------------------------------------------
    var percent = 0;
    // 页面总高
    var totalH =
      document.body.scrollHeight || document.documentElement.scrollHeight;

    // 可视高
    var clientH = window.innerHeight || document.documentElement.clientHeight;

    window.addEventListener('scroll', function (e) {
      // 计算有效高
      var validH = totalH - clientH;

      // 滚动条卷去高度
      var scrollH =
        document.body.scrollTop || document.documentElement.scrollTop;

      // 百分比
      var result = (scrollH / validH).toFixed(1);

      percent = percent > result ? percent : result; // 取用户滑动的最大百分比0-1

      console.log(result, percent);
    });

    // 3.计算停留时间-------------------------------------------------------------------------
    let realTimes = 0;
    let timer = null;
    window.addEventListener('pageshow', () => {
      var count = 0;
      timer = null; //timer变量记录定时器setInterval的返回值
      timer = setInterval(function () {
        realTimes = ++count;
        document.getElementById('count').innerHTML = count;
      }, 1000);
    });

    // 4.计算图片长按次数---------------------------------------------------------------------
    let touchLongTimes = 0;
    // dom 为你需要绑定的节点，三个事件绑定到同一个节点上，比如背景、地图等节点
    let dom = document.getElementById('img');
    touchLong(dom);

    function touchLong(dom) {
      let timeOutEvent = 0; // 用来设置定时器，且 0 数值用于后续判断
      dom.addEventListener('touchstart', function (e) {
        // 阻止它的默认事件
        e.preventDefault();
        // 当开始触摸的时候定时器计时
        timeOutEvent = setTimeout(function () {
          timeOutEvent = 0;
          console.log('你长按了');
          touchLongTimes++;
        }, 1000); // 达到 1s 就会实现长按效果
      });
      dom.addEventListener('touchmove', function (e) {
        // 如果触摸未达到 1s 且开始移动，则清除计时器
        clearTimeout(timeOutEvent);
        timeOutEvent = 0;
      });
      dom.addEventListener('touchend', function (e) {
        // 如果触摸未达到 1s 且离开屏幕，则清除计时器
        clearTimeout(timeOutEvent);
        if (timeOutEvent !== 0) {
          // 判断是否发生长按
          console.log('你这是点击，不是长按');
        }
        return false;
      });
    }

    // 左滑关闭
    // 获取触摸起始位置
    var startX;
		var startY;
    document.addEventListener('touchstart', function (event) {
      startX = event.touches[0].pageX;
      startY = event.touches[0].pageY;
    });

    // 监听触摸移动事件
    document.addEventListener('touchmove', function (event) {
      var moveX = event.touches[0].pageX;
      var moveY = event.touches[0].pageY;
			let X = moveX - startX;
      let Y = moveY - startY;

      // 判断是否为左滑事件
      if (Math.abs(X) > Math.abs(Y) && X > 0 && Math.abs(Y) < 9) {
         console.log('左滑了');
        // 执行左滑事件处理
          aa();
      }
    });
   
    // 全局变量
    var sendTimes = 1;
    function aa(params) {
      if (sendTimes == 0) return;
      sendTimes = 0;

      clearInterval(timer);
      console.log(
        '已经在该页面停留' +
          realTimes +
          '秒了，' +
          '查看的百分比：' +
          percent +
          ',用户滑动页面次数：' +
          slider +
          ',长按图片次数为：' +
          touchLongTimes +
          '，我要退出页面啦 '
      );
      // 测试接口
      // let parames = {
      //     realTimes,
      //     percent,
      //     slider,
      //     touchLongTimes,
      //     params
      // };
      let parames = {
        a: 1,
      };
      // https://developer.mozilla.org/zh-CN/docs/Web/API/Navigator/sendBeacon
      let res = navigator.sendBeacon(
        'https://zebinlin.cn/api/helloworld',
        JSON.stringify(parames)
      );
    }

    //页面离开的时候--发送请求--------------------------------------------------------------------
    window.addEventListener('pagehide', () => {
      aa();
    });
  </script>
</html>
