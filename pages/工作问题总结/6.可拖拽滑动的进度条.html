    可拖拽滑动的进度条-组件

    Ada 2023/5/10

    vue.文件-可参考的一种思路，效果如下 

    <template>
  <div class="slider" ref="slider">
    <div class="process" :style="{ width }"></div>
    <div class="thunk" ref="trunk" :style="{ left }">
      <div class="block"></div>
      <!-- <div class="tips">
        <span>{{ scale * 100 }}</span>
        <i class="fas fa-caret-down"></i>
      </div> -->
    </div>
  </div>
</template>
 <script>
/*
 * min 进度条最小值
 * max 进度条最大值
 * v-model 对当前值进行双向绑定实时显示拖拽进度
 * */
export default {
  props: ["min", "max", "percent"],

  data() {
    return {
      slider: null, //滚动条DOM元素
      thunk: null, //拖拽DOM元素
      tempPercent: 0, //当前值
    };
  },
  watch: {
    // 此处监听variable变量，当期有变化时执行
    percent(val) {
      this.tempPercent = val;
      this.init();
    },
  },
  //渲染到页面的时候
  mounted() {
    this.init();
  },
  computed: {
    // 设置一个百分比，提供计算slider进度宽度和trunk的left值
    // 对应公式为 当前值-最小值/最大值-最小值 = slider进度width / slider总width
    // trunk left = slider进度width + trunk宽度/2
    scale() {
      return (this.tempPercent - this.min) / (this.max - this.min);
    },
    width() {
      if (this.slider) {
        return this.slider.offsetWidth * this.scale + "px";
      } else {
        return 0 + "px";
      }
    },
    left() {
      if (this.slider) {
        return (
          this.slider.offsetWidth * this.scale -
          this.thunk.offsetWidth / 2 +
          "px"
        );
      } else {
        return 0 + "px";
      }
    },
  },
  methods: {
    init() {
      this.slider = this.$refs.slider;
      this.thunk = this.$refs.trunk;
      var _this = this;
      this.thunk.onmousedown = function (e) {
        var width = parseInt(_this.width);
        var disX = e.clientX;
        document.onmousemove = function (e) {
          // value, left, width
          // 当value变化的时候，会通过计算属性修改left，width

          // 拖拽的时候获取的新width
          var newWidth = e.clientX - disX + width;
          // 拖拽的时候得到新的百分比
          var scale = newWidth / _this.slider.offsetWidth;
          _this.tempPercent = Math.ceil(
            (_this.max - _this.min) * scale + _this.min
          );
          _this.tempPercent = Math.max(_this.tempPercent, _this.min);
          _this.tempPercent = Math.min(_this.tempPercent, _this.max);
          _this.$emit("percentChange", _this.tempPercent);
        };
        document.onmouseup = function () {
          document.onmousemove = document.onmouseup = null;
        };
        return false;
      };
    },
  },
};
</script>
