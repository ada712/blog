//遍历比较，设置一个最长的连续序列长度和一个当前遍历最长连续的长度
var longestConsecutive = function(nums) {
    // 临界条件
    if(nums.length == 0) return 0;

    let maxlen = 0 //最长连续序列长度
    let currlen = 1 //当前最长连续序列长度
    nums.sort((a,b) => a-b) //升序排序

    for(let i=0;i<nums.length;i++){
        if((nums[i] - 1) > nums[i-1]){//当出现不连续时，将currlen值初始化
            currlen =  1
        }
        else if((nums[i] - 1) == nums[i-1]){ //连续时，currlen值连续加1
            currlen = currlen + 1
        }
        maxlen = currlen > maxlen ? currlen : maxlen;
    }
    return maxlen
};

    # 15.三数之和（排序+双指针+去重） (opens new window)

    给你一个包含 n 个整数的数组 nums，判断 nums 中是否存在三个元素 a，b，c ，使得 a + b + c = 0 ？请你找出所有和为 0 且不重复的三元组。

注意：答案中不可以包含重复的三元组。

### 问题分析

如果只寻找满足条件的三元组其实是很容易的，我们可以分步操作，先计算两数之和，在判断是否有第三个数加上满足和为零，记录满足条件的三元组，但是这样会产生很多重复的三元组，所以需要对三元组进行去重，一般的方法处理会显得非常繁杂，不好操作。

这里使用**双指针**是比较容易操作的

基本操作流程为：

- 先给数组以升序排好序
- 外层一个 for 循环遍历数组，i 对应其下标
- 循环内部定义两个指针 left=i+1，right=nums.length-1，其中 a=nums[i]，b=nums[left]，c=nums[right]
- 通过移动两个指针找到当前 a=nums[i]时，所有满足条件的三元组
- 移动方式为：
  当 nums[i] + nums[left] + nums[right] > 0 时，说明三数之和大了，right 左移
  当 nums[i] + nums[left] + nums[right] < 0 时，说明三数之和小了，left 右移

    var threeSum = function (nums) {
  const len = nums.length;
  // 结果集
  let result = [];
  // 思路： 排序 + 双指针
  nums.sort((a, b) => a - b);

  //因为是排好序，从小到大，如果a大于零，表示不存在满足条件的三元组
  if (nums[0] > 0) return result;

  for (let i = 0; i < len; i++) {
    // a去重
    if (i > 0 && nums[i - 1] == nums[i]) continue;
    let l = i + 1,
      r = len - 1;
    while (l < r) {
      let total = nums[i] + nums[l] + nums[r];
      if (total > 0) r--;
      if (total < 0) l++;
      if (total == 0) {
        result.push([nums[i], nums[l], nums[r]]);
        l++;
        r--;
        // b,c去重- 这里主要是不需要重复的满足条件的res
        while (l < r && nums[l] == nums[l - 1]) l++;
        while (l < r && nums[r] == nums[r + 1]) r--;
      }
    }
  }
  return result;
};

    var lengthOfLongestSubstring = function (s) {
  let left = 0, max = 0; // left 记录无重复子串--开始--下标
  let map = new Map();
  for (let right = 0; right < s.length; right++) {
    // 当哈希表中存在当前元素并且位置在左边界右边，将left移动到哈希表当前元素位置+1
    if (map.has(s[right]) && left <= map.get(s[right])) {
      left = map.get(s[right]) + 1;
    }
    // 存在相同元素，更新位置
    map.set(s[right], right);
    max = Math.max(max, right - left + 1);
  }
  return max;
};

    const findAnagrams = (s, p) => {
  const need = new Map();
  const window = new Map();
  // 统计p中字符的个数
  for (const char of p) {
    need.set(char, (need.get(char) || 0) + 1);
  }
  // 定义左右指针
  let [left, right] = [0, 0];
  // 达到要求的字符数
  let valid = 0;
  const res = [];
  const lenS = s.length;
  const lenP = p.length;
  while (right < lenS) {
    // 右指针对应的字符，右指针右移
    const c = s[right++];
    if (need.get(c)) {
      // 若在need中，window中相应的+1
      window.set(c, (window.get(c) || 0) + 1);
      // 若window中，此字符达到了need中的数量，说明此字符满足要求
      if (window.get(c) === need.get(c)) valid++;
    }
    while (right - left == lenP) {
      // 若valid等于need的大小，说明need中每个字符都满足要求了
      // 把左指针放入res
      if (valid === need.size) res.push(left);
      // 左指针对应的字符，左指针右移
      const d = s[left++];
      if (need.get(d)) {
        // 若出窗口的字符在need中，肯定不满足条件了，valid--
        if (window.get(d) === need.get(d)) valid--;
        // 将window中相应字符数-1,这一步要放判断后面
        window.set(d, window.get(d) - 1);
      }
    }
  }
  return res;
};

    # 思路

    prev 初始为第一个区间，cur 表示当前的区间，res 表示结果数组

    • 开启遍历，尝试合并 prev 和 cur，合并后更新到 prev
    • 合并后的新区间还可能和后面的区间重合，继续尝试合并新的 cur，更新给 prev

    # 合并的策略

    原则上要更新 prev[0]和 prev[1]，即左右端:

    • prev[0] = min(prev[0], cur[0])
    • prev[1] = max(prev[1], cur[1])
    • 但如果先按区间的左端排升序，就能保证 prev[0] < cur[0]
    • 所以合并只需这条：prev[1] = max(prev[1], cur[1])

    易错点

    我们是先合并，遇到不重合再推入 prev。 当考察完最后一个区间，后面没区间了，遇不到不重合区间，最后的 prev 没推入 res。 要单独补上。

// 算法步骤
// 从右往左找到第一个比其后一个元素小的元素 nums[k]。
// 从右往左找到第一个比 nums[k] 大的元素 nums[l]。

// 将 nums[k+1:] 子数组逆序排列。
// [1,2,4,3] ,这个来举例子你就好懂了

function nextPermutation(nums) {
    // 从右往左找到第一个比其后一个元素小的元素
    let k = nums.length - 2;
    while (k >= 0 && nums[k] >= nums[k + 1]) {
        k--;
    }

    // 如果整个数组是降序排列，则直接翻转整个数组
    if (k < 0) {
        nums.reverse();
        return;
    }

    // 从右往左找到第一个比 nums[k] 大的元素
    let l = nums.length - 1;
    while (nums[l] <= nums[k]) {
        l--;
    }

    // 交换 nums[k] 和 nums[l]
    [nums[k], nums[l]] = [nums[l], nums[k]];

    // 将 nums[k+1:] 子数组逆序排列
    let remaining = nums.splice(k + 1);
    remaining.reverse();
    nums.push(...remaining);
};
