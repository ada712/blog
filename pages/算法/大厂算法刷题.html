    # 1.绳子最大覆盖点数【窗口】

    描述：
给定一个有序数组arr，arr中的每个值代表坐落在x轴上的点，再给定一个正整数K，表示一根绳子的长度，
绳子可以任意放在x轴上，请返回绳子最多能盖住多少个点。注意绳子的边缘盖住也算作盖住。

    const maxCoverPoint=(arr,k)=>{
  //双指针 时间复杂度为O(N)
  if (arr == null || arr.length == 0 || K <= 0) {
      return 0;
  }
  let R=0,max=0;
  for(let L=0;L<arr.length;L++){
      while(R < arr.length && arr[R] - arr[L] <= K){
          R++;
      }
      //R指向的是匹配失败的位置
      max=Math.max(max,R-L);
  }
  return max;
}

    # 2.相邻字符的交换次数

    描述：
一个数组中只有两种字符’G’和’B’，可以让所有的G都放在左侧，所有的B都放在右侧。
或者可以让所有的G都放在右侧，所有的B都放在左侧。但是只能在相邻字符之间进行交换操作。
返回至少需要交换几次。
    1
    2
    3
    4
    核心：把G全部移动到左边，记录次数step1；把B全部移动到左边，记录次数step2。返回值取step1和step2的最小值。

const minSteps = (s) => {
  if(s.length == 0) return 0;
  let step1 = 0, step2= 0;
  let g = 0, b = 0; // 左指针位置

  for(let i=0; i<s.length; i++){
    if(s[i] == 'G'){ // 方案1：当前的g去左边
      step1 = step1 + i - (g++);
    }else{           // 方案2：当前的b去左边
      step2 = step2 + i - (b++);
    }
  }
  return Math.min(step1,step2);
}

    var findTargetSumWays = function(nums, S) {
    // 方法1：深度遍历
    let count = 0;
    function dfs(i, res) {
        if(i === nums.length) {
            if(res === S) {
                count++;
            }
            return;
        };
        dfs(i+1, res+nums[i]);
        dfs(i+1, res-nums[i]);
    }
    dfs(1, nums[0]);
    dfs(1, -nums[0]);
    return count;
};

    const maxPairNums  = function(arr,k){
  const n = arr.length;
  if(arr == null || k<0 || arr.length<2) return 0;
  arr.sort();
  let l = 0;
  let r = 0;
  let used = []; // 用于记录有没有用过

  while(l<n && r<n){
    if(used[l]){ // 被用了，则直接++
      l++;
    }else if(l>=r){ // 说明没有了，或者只剩下1个
      r++;
    }else{ // 不止一个数，而且都没有用过

        res++;
        used[r++]=true;
        l++;
      }else if(distance < k){
        r++;
      }else{
        l++;
      }
    }
  }
  return res;
}

    const maxNums  = function(arr,limit){
  // 先排序
  arr.sort();
  let l = 0;
  let r = arr.length - 1;
  let res = 0;
  while(arr[r]>= limit){
    res++;
    r--;
  }
  //此时【l,r】该区间的所有数都<limit;在该区间找2数之和小于limit
  while(l<r){
    let sum = arr[l]+arr[r];
    if(sum > limit){
      res++;
      r--;
    }
    if(sum <= limit){
      res++;
      l++;
      r--;
    }
  }
  return  l== r? res+1: res; // 判断是剩下1还是刚好用完
}

    let pre = 0, maxAns = nums[0];
