    # 实现一个自适应的卡片列表

    考察的知识点：
flex属性 是 flex-grow、flex-shrink、flex-basis三个属性的缩写。

flex-grow：定义项目的的**放大**比例；

默认为0，即 即使存在剩余空间，也不会放大；
所有项目的flex-grow为1：等分剩余空间（自动放大占位）；
flex-grow为n的项目，占据的空间（放大的比例）是flex-grow为1的n倍。

flex-shrink：定义项目的**缩小**比例；

默认为1，即 如果空间不足，该项目将缩小；
所有项目的flex-shrink为1：当空间不足时，缩小的比例相同；
flex-shrink为0：空间不足时，该项目不会缩小；
flex-shrink为n的项目，空间不足时缩小的比例是flex-shrink为1的n倍。

flex-basis： 定义在分配多余空间之前，项目占据的主轴空间（main size），
浏览器根据此属性计算主轴是否有多余空间，

默认值为auto，即 项目原本大小；
设置后项目将占据固定空间。
所以flex属性的默认值为：0 1 auto （不放大会缩小）
flex为none：0 0 auto （不放大也不缩小）
flex为auto：1 1 auto （放大且缩小）

flex：1 即为flex-grow：1，经常用作自适应布局，将父容器的display：flex，侧边栏大小固定后，将内容区flex：1，内容区则会自动放大占满剩余空间。

    • 函数柯里化是一种技术，一种将多入参函数变成单入参函数

    所谓的函数柯里化 currying，最主要的场景就是减少不必要的固定入参。是函数的一种拓展技术

    function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn(...args); // 接受完所有参数，直接执行
    } else {
      return function (...args2) {
        // 未接收完参数
        return curried(...args,...args2);
      };
    }
  };
}

    class LazyMan {
  constructor(name) {
    this.name = name;
    this.task = []; // 任务队列
    console.log(`My named ${name}`);

    // 这里使用异步调用next()是为了确保所有链式调用都被添加到task[]才开始执行任务
    setTimeout(() => {
      this.next();
    });
  }

  sleep(time) {
    this.task.push(() => {
      console.log(`I am sleeping...`);
      setTimeout(() => {
        console.log(`after ${time} ms`);
        this.next();
      }, time);
    });
    return this;
  }

  eat(food) {
    this.task.push(() => {
      console.log(food);
      this.next();
    });
    return this;
  }

  next() {
    let fn = this.task.shift();
    fn && fn(); // if(fn) fn()
  }
}

    # 小牛试刀

    • 要求如下

    详细描述:

    实现一个类 Job，提供以下接口：

      1. task() 方法，执行一次任务，打印 'task',
      1. cancel() 方法， 取消上一次任务的执行，成功取消打印 'cancel'， 如果未有任务可以取消，打印 'notask'
      1. sleep(ms: number) 方法，ms 毫秒后再执行后面的任务，该任务本身可以被 cancel
      1. 所有接口支持链式调用
    class Job {
  constructor() {
    this.tasks = [];
    // 这里使用异步调用next()是为了确保所有链式调用都被添加到tasks[]才开始执行任务
    setTimeout(() => {
      this.next();
    }, 0);
  }

  task() {
    this.tasks.push(() => {
      console.log('task');
      this.next();
    });
    return this;
  }

  cancel() {
    // 先判断再push

    if (this.tasks.length == 0) {
      this.tasks.push(() => {
        console.log('notask');
        this.next();
      });
    } else {
      this.tasks.pop();
      this.tasks.push(() => {
        console.log('cancle');
        this.next();
      });
    }
    return this;
  }

  sleep(ms) {
    this.tasks.push(() => {
      // console.log(`I am sleeping...`);
      setTimeout(() => {
        // console.log(`after ${ms} ms`);
        this.next();
      }, ms);
    });
    return this;
  }

  next() {
    let fn = this.tasks.shift();
    fn && fn();
    // if(fn) fn()
  }
}

    # 实现大数相加

    js超过9007199254740991被称为不安全的大数，其获取以及运算都会出现精度丢失，
那么如果有两个数是大数（string类型），请实现这两个数相加而不丢失精度(string类型)

输入描述

两个大数，或者不是大数，比如‘9007199254740991’, ‘2’

输出描述

两个数字相加的结果，要求不丢失精度，‘9007199254740991’ + ‘2’ = ‘9007199254740993’而不是9007199254740992

    function add(a, b) {
  let result = '';
  let len = Math.max(a.length, b.length);

  // ES2017 引入了字符串补全长度的功能。如果某个字符串不够指定长度，
  // 会在头部或尾部补全。padStart()用于头部补全，padEnd()用于尾部补全。
  a = a.padStart(len, '0');
  b = b.padStart(len, '0');
  let isCarry = false;

  for (let i = len - 1; i >= 0; i--) {
    let temp = Number(a[i]) + Number(b[i]);
    isCarry && temp++;
    isCarry = temp > 10;
    temp = temp > 10 ? temp - 10 : temp;
    result = temp + result;
  }
  if (isCarry) {
    result = 1 + result;
  }
  return result;
}

    # JavaScript 实现异步任务调度器

    实现一个 Scheduler 类，使下面的代码能正确输出

    // 延迟函数
const sleep = (time) => new Promise((resolve) => setTimeout(resolve, time));

// 同时进行的任务最多2个
const scheduler = new Scheduler(2);

    关于异步并发调度器关键实现：

    • 需要有一个数组 queue，模拟队列（先进先出），依次进行异步请求处理。
    • add 方法用来添加异步请求；根据当前正在 running 的个数，判断添加异步请求是执行 run 还是将其放入到队列中。
    • run 方法真正执行异步请求
    class Scheduler {
  constructor(max) {
    this.max = max; // 限制个数
    this.queue = []; // 先进先出队伍
    this.running = 0; // 正在运行的个数
  }

  add(task) {
    return new Promise((resolve) => {
      // 关键，加了一层 Promise 的封装，将其 resolve 的函数交出去
      task.resolve = resolve;

      if (this.running < this.max) {
        this.run(task);
      } else {
        this.queue.push(task);
      }
    });
  }
  async run(task) {
    this.running++;
    await task(); // await 后面的代码相当于放在then 里面
    task.resolve(); //task异步任务完成以后，再调用外层Promise的resolve以便add().then()的执行
    this.running--;
    if (this.queue.length > 0) {
      const fn = this.queue.shift();
      this.run(fn);
    }
  }
}

    # 对金额数字进行千分符格式化处理

    实现一个formatNumber 方法， 接受一个数字，返回一个【以千分位分隔】的字符串，要求支持正数、小数、负数等场景
    1
    • 原生 js 实现
    function formatNumberWithCommas(number) {
  // 类型检测
  if (typeof num !== 'number') {
    return '';
  }
  // 先转成字符串
  number += '';
  // 支持小数,按小数点分成两部分  使用了es6解构
  let [integer, decimal] = number.split('.');
  // 封装了doSplit方法 第二个参数isInteger来表示是整数部分还是小数部分
  const doSplit = (num, isInteger = true) => {
    // 如果为空，直接返回
    if (num === '') return '';
    // 如果是整数部分  先按位切割再返转
    // 整数部分数字从右往左数，每3位插入一个逗号
    // 小数部分从左往右数
    // 两次反转，它的逗号顺序是一样的。
    if (isInteger) num = num.split('').reverse();
    let str = [];
    for (let i = 0; i < num.length; i++) {
      if (i !== 0 && i % 3 === 0) str.push(',');
      str.push(num[i]);
    }
    if (isInteger) return str.reverse().join('');
    return str.join('');
  };
  integer = doSplit(integer);
  decimal = doSplit(decimal, false);
  return integer + (decimal === '' ? '' : '.' + decimal);
}

    function formatNumberWithLocale(number) {
  let [integer, decimal = ''] = (number + '').split('.');
  // 1,234,567 整数部分可通过toLocalString实现
  integer = (+integer).toLocaleString();
  if (decimal === '') return integer;
  // 小数部分另外处理
  decimal = decimal.split('').reverse().join('');
  decimal = (+decimal).toLocaleString();
  decimal = decimal.split('').reverse().join('');
  return integer + '.' + decimal;
}
