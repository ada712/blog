    class EventBus {
  constructor() {
    this.listeners = {};
  }

  on(type, cb) {
    if (!this.listeners[type]) {
      this.listeners[type] = [];
    }
    this.listeners[type].push(cb);
  }
  emit(type, ...args) {
    if (this.listeners[type] && this.listeners[type].length > 0) {
      const types = this.listeners[type];
      types.forEach((cb) => cb(...args));
    }
  }
  off(type, cb) {
    if (this.listeners[type]) {
      const curIndex = this.listeners[type].findIndex((it) => it === cb);
      if (curIndex >= 0) {
        this.listeners[type].splice(curIndex, 1);
      }
      if (this.listeners[type].length === 0) {
        delete this.listeners[type];
      }
    }
  }
}

    # 二叉树的堂兄弟节点---3

    详细描述

    在二叉树中，根节点位于深度 0 处，每个深度为 k 的节点的子节点位于深度 k+1 处。

    如果二叉树的两个节点深度相同，但 父节点不同 ，则它们是一对堂兄弟节点。

    我们给出了具有唯一值的二叉树的根节点 root ，以及树中两个不同节点的值 x 和 y。

    只有与值 x 和 y 对应的节点是堂兄弟节点时，才返回 true 。否则，返回 false。

    输入输出描述

    输入描述

    (root = [1, 2, 3, 4]), (x = 4), (y = 3);
    1

    输出描述

    返回 false，3 和 4 不是堂兄弟节点

    题目解析

    采用广度优先搜索，使用 hasX 和 hasY 标识是否已经出现过 x 和 y。

    • 用一个栈维护同一深度的节点
    • 逐个取出栈中的节点，获取其左右节点，考虑以下情况
      • 对于左节点
        • 如果左节点的值为 x，假设 hasY 为 true，直接返回 true；否则将 hasX 设为 true
        • 如果左节点的值为 y，假设 hasX 为 true，直接返回 true；否则将 hasY 设为 true
      • 对于右节点
        • 右节点的判断逻辑和左节点几乎一样，但是需要加入判断当右节点的值为 x 或者 y 时，左节点（兄弟节点）的值是否为 y 或者 x，如果是，则返回 false
    • 遍历一个深度，如果 hasX 或者 hasY 为 true，停止遍历，返回 false；否则回到第二步

    ListNode* reverse(ListNode *head, ListNode *end) {
	// 设置递归结束条件
    if (head == end) return end;
    // 先递归，后执行逻辑。相当于从后往前操作链表
    ListNode *last = reverse(head->next, end);

    return last;

ListNode* reverseKGroup(ListNode* head, int k) {

    ListNode *newHead = reverse(head, end);
    // 递归执行下一个 k 组
    head->next = reverseKGroup(temp, k);
    return newHead;
}

    # 需求疲劳度最小值---4

    详细描述

    Leader 在和产品开完高阶会议后，现在在做重构的人数安排工作，leader 需要安排 d 天的工作，即是工作必须在 d 天完成，不能提前和延期。

    leader 现在手上有 n 个工作需要完成，每天至少分配一个工作，而且每个工作都可以在一天内完成。每个工作都有一个相对应的疲劳度，疲劳度存在一个数组里面，如 tired = [6, 5, 4, 3, 2, 1]，代表完成工作 i 需要用到 tired[i] 个疲劳度，完成某个工作后稍作休息后又可以继续选择做下一个工作或不做任何工作，当天组员的疲劳度为当天完成的工作中的最大疲劳度。

    现在要求到整个工作计划最小的疲劳度总数。如果无法制定计划，则返回 -1。

    限制条件

    • 1 <= tired.length <= 300
    • 1 <= tired[i] <= 1000
    • 1 <= d <= 10

    例子：

    输入：tired = [5, 4, 3, 2, 1], d = 2

    输出：6

    解释：第一天安排去完成前4项工作**,** 疲劳度为5；第二天安排去完成最后一个工作，疲劳度为1，总的疲劳度为6）。

    例子：

    输入：tired = [1, 1], d = 3

    输出：-1

    解释：即使每天安排去完成一个工作，仍会有一天空余出来，不满足题意（题意是工作，必须在 d 天完成，不能提前和延期。）

    输入输出描述

    输入描述

    输入：tired = [5, 4, 3, 2, 1], d = 2

    tired 代表的是每个工作需要的疲劳度，d 代表的是需要在 d 天完成全部工作。

    输出描述

    输出：6

    整个工作计划的最小疲劳度总数。

    题目解析

    思路：

    问题可以简化为，对一个数组分成 d 个子数组，求出每个子数组的最大值，把各个子数组的最大值求和，在各种分法中，求出其中的最小值。求解状态最大最小值的时候，很多时候会想到使用动态规划。

    现在我们用 dp[i][j] 表示前 i 天完成 j 个工作的最小疲劳度，而且当 j >= i 时才会有求解值，否则为 -1。

    对于 dp[i][j], 这时候 i 天完成 j 个工作，我们有多种选择，可以看成是前 i-1 天完成 k-1 项任务(因为每天至少完成一项所以，k-1>=i-1)，然后第 i 天完成第 k,k+1,...,i 项任务，即 即 dp[i][j]=min{dp[i-1][k-1] + max{jobDifficulty[k],jobDifficulty[k+1],...,jobDifficulty[i]}},i <= k <=jobDiffculty.size()

    题解：

    class Solution {
  run(tired, d) {
    let n = tired.length;
    if (n < d) return -1;
    let dp = Array.from({ length: d + 1 }, () =>
      Array.from({ length: n + 1 }, () => -1)
    );

    dp[1][1] = tired[0];
    // 初始化 1 天完成 1 个任务，2 个任务 ..... n 个任务。疲劳度自然就是这些任务中的最大疲劳值。
    for (let i = 2; i < n + 1; i++) {
      dp[1][i] = Math.max(dp[1][i - 1], tired[i - 1]);
    }

    for (let i = 2; i < d + 1; i++) {
      for (let j = i; j < n + 1; j++) {
        // 初始化是前 i-1 天做了 j-1 个工作，最后一天做最后一个工作
        dp[i][j] = dp[i - 1][j - 1] + tired[j - 1];
        // numTired 是 工作 k 到 j 最大的疲劳值
        let numTired = tired[j - 1];
        for (let k = j - 2; k > i - 2; k--) {
          numTired = Math.max(tired[k], numTired);
          if (dp[i - 1][k] + numTired < dp[i][j]) {
            // i-1 天做前 k 个工作，最后一天做 k 到 j 的工作
            dp[i][j] = dp[i - 1][k] + numTired;
          }
        }
      }
    }
    return dp[d][tired.length];
  }
}

    解释：每天都只做一个需求，每个需求都找导师求助，这样做需求最多的一天花费的时间数 t 是 Math.ceil(5 / 2) = 3。

    输入描述

    输入：t = [1, 2, 4, 3, 5], a = 2

    数组 t 代表的是完成各个任务的时间数（就是一个时间度量，不是天数），a 代表的是需要完成任务最多的天数。

    题目解析

    思路：

    问题可以简化为把一个数组分成若干个子数组，子数组数量小于或等于 a，有若干种分法，在每一种分法里，把每个子数组的最大值除以 2，并向上取整，再求每个子数组之和，，取其中的最大值 t，在这若干种分法里，求 t 的最小值。

    如果不考虑向导师请求帮助。假设某个子数组之和为 t 满足条件，那么 t + 1, t + 2, …… 也是满足的条件的，求其中的 t 的最小值。所以直接 二分查找 就可以找到满足条件的 最小 t 值。

    那现在我们再来看，现在可以在每个子数组里面找到一个最大值，并对此值除以 2 并向上取整，在计算每个子数组之和时，给定一个估计的 limit （即为子数组之和的最大值），维护一个当前序列的最大值，然后求：

    当前序列之和 - 最大值除以 2 并向下取整

    ，（因为我们要保留最大值除以 2 并向上取整）该值不能大于 limit。 最后判断子数组的数量是否小于或等于 a，是的话说明当前的估计的 limit 是有效的 t，接着可以尝试减少limit的大小看看结果是否还符合条件；不是的话就增大 limit 的大小，直到找到符合条件的 limit。

    题解：

    class Solution {
  run(t, a) {
    let start = 0;
    let end = t.reduce((pre, cur) => pre + cur);
    while (start + 1 < end) {
      let mid = start + Math.ceil((end - start) / 2);
      if (this.check(t, a, mid)) {
        end = mid;
      } else {
        start = mid + 1;
      }
    }
    if (this.check(t, a, start)) return start;
    return end;
  }

  check(t, a, limit) {
    for (let i in t) {
      // 如果某个需求的时间数除以2并向上取整的值，比 limit 还高的话，说明 limit 不满足条件，这时候需要增大 limit 的值，直接返回 false
      if (Math.ceil(t[i] / 2) > limit) {
        return false;
      }
    }
    let useDay = 1;
    let curSum = t[0];
    let curMax = t[0];
    for (let i = 1; i < t.length; i++) {
      // 因为要取当前序列的最大值的一半，并对此向上取整，
      // curSum 是所有原值之和，减去最大值的一半，并且此值需要向下取整
      if (curSum + t[i] - Math.floor(Math.max(curMax, t[i]) / 2) <= limit) {
        curSum += t[i];
        curMax = Math.max(curMax, t[i]);
      } else {
        useDay += 1;
        curSum = t[i];
        curMax = t[i];
      }
    }
    return useDay <= a;
  }
}

  const origin = head
  if (!head) {
    return null
  }
  let count = 1
  while (head.next) {
    count++
    head = head.next
  }
  head = origin

    const index = count - n - 1

    function quickSort(array) {
  // 当 array 的长度小于等于 1，直接返回 array 即可。
  if (array.length > 1) {
    // helper 数组用来模拟递归栈，里面的每一项表示待排序的 index 范围。
    // 初始值为 [0, array.length-1]
    const helper = [[0, array.length - 1]];
    // 循环 helper.length 直到 length 等于 0 结束。
    while (helper.length) {
      // 推出待排序的 index 范围。
      const toSort = helper.shift();
      // 取出 leftIndex 和 rightIndex。
      const [leftIndex, rightIndex] = toSort;
      // 左指针 left 初始值为 leftIndex，右指针 right 初始值为 rightIndex。
      // pivot 基准位设成中间位。
      let left = leftIndex,
        right = rightIndex,
        pivot = array[Math.floor((leftIndex + rightIndex) / 2)];
      // 循环直到左指针大于右指针结束。
      while (left <= right) {
        // 当左指针指向的项小于基准位项，则左指针往右移动一位。
        while (array[left] < pivot) {
          left++;
        }
        // 当右指针指向的项大于基准位项，则右指针往左移动一位。
        while (array[right] > pivot) {
          right--;
        }
        // 如果左指针小于等于右指针，交换两个指针指向的项，并且分别往左和右移动一位。
        if (left <= right) {
          [array[left], array[right]] = [array[right], array[left]];
          left++;
          right--;
        }
      }
      // 如果 leftIndex 小于 left-1，则把对应的 index 范围推进 helper 函数中。
      if (leftIndex < left - 1) {
        helper.push([leftIndex, left - 1]);
      }
      // 如果 rightIndex 大于 left，则把对应的 index 范围推进 helper 函数中。
      if (left < rightIndex) {
        helper.push([left, rightIndex]);
      }
    }
  }
  return array;
}
